generativeText examples

Basic example

Randomly select a different font-family for each letter. Will produce a different result on each load of the page.

Nevermind the Bollocks, here's generativeText!

Code


var params = {
	fontFamily: {
		values: [
			"'Alegreya Sans SC', sans-serif",
			"'Exo 2', sans-serif",
			"'Oleo Script Swash Caps', cursive",
			"'IM Fell Great Primer', serif",
			"'Gilda Display', serif",
			"'Macondo Swash Caps', cursive",
			"'PT Mono', sans-serif",
			"'Chicle', cursive",
			"'Sigmar One', cursive",
			"'Lily Script One', cursive",
			"'Denk One', sans-serif",
			"'Montaga', serif",
			"'Mystery Quest', cursive",
			"'Alike Angular', serif"
		]
	}
};

var ex1 = new generativeText(params);
ex1.applyToElementById("fontFace-example");

Apply to words

Change background and foreground color by repeated steps and apply it to the words on the text.

This reminds me of 90's Wired Mag for some reason.

Code


var params2 = {
	backgroundColor: {
		type: 'list',
		values: [
			'#ca3a11',
			'#333333',
		],
		steps:true,
	},
	color: {
		type: 'list',
		values: [
			'#333333',
			'#ca3a11',
		],
		steps:true,
	}
};

var opts2 = {
	applyTo: "words",
	textSpaces: 'all',
}

var ex2 = new generativeText(params2, opts2);
ex2.applyToElementById("example3");

Using Step functions

You can use all kind of weird functions you want on the parameters.

This text will bend into a weird and cool shape. Isn't this pretty amazing? I'm just adding some more text here so that in the next example I can show you the difference bettween using applyTo: "text" and applyTo: "wrapped".

Code


var params4 = {
	fontSize: {
		type:'generate',
		unit:'em',
		min:0.4,
		max:1,
		steps:true,
		stepFunction: function (f) {
			two_pi = 6.2832;
			step = two_pi / this.totalSteps;
			val = (Math.cos( (step * this.currentStep ) ) +1) / 2;
			return this.min + (this.range * val);
		}
	}
};

var ex4 = new generativeText(params4);
ex4.applyToElementById("example4");

applyTo: "wrapped"

As you probably saw on the last example, by default, generativeText text breaks lines to fit the container width. This means that a line break may occur anywhere on the word. This can be prevented by using applyTo: "wrapped"

Usually used in reference to a computer application, a text-based application is one whose primary input and output are based on text rather than graphics or sound. This does not mean that text-based applications do not have graphics or sound, just that the graphics or sound are secondary to the text.

Code


var params5 = {
	color: {
		type:'generate',
		r: {
			min: '00',
			max: 'AA',
			steps: true
		},
		g: {
			min: '00',
			max: 'AA',
			steps: true
		},
		b: {
			min: '66',
			max: 'AA',
			steps: true
		}
	}
};

var opts5 = {
	applyTo: "wrapped",

ex5.applyToElementById("example5");

Using Post or Pre functions

You can use Post or Pre functions to each step of the process. This P functions will be will be run before or after each element is processed.

Usually used in reference to a computer application, a text-based application is one whose primary input and output are based on text rather than graphics or sound. This does not mean that text-based applications do not have graphics or sound, just that the graphics or sound are secondary to the text.

Code


var params6 = {
	color: {
		type:'list',
		values: ['ff0000', '000000', 'efefef'],
	}
};

var conditionalBackground = {

	values: ['ff0000', '000000', 'eeeeee'],

	posFunc: function() {

		var current = this.memory[ this.currentStep ];
		var color = current.style.color;
		var rand = Math.round(Math.random());
		switch(color) {
			case "rgb(255, 0, 0)":
				var key = 1 + rand;
				break;
			case "rgb(0, 0, 0)":
				var key = (2 + rand) % 3;
				break;
			case "rgb(239, 239, 239)":
				var key = rand;
			break;
		}
		current.style.backgroundColor = "#" + this.values[ key ];
	}
};

var opts6 = {
	applyTo: 'words',
	pObj: conditionalBackground,
	memory: true,
	textSpaces: "both",
};

var ex6 = new generativeText(params6, opts6);
ex6.applyToElementById("example6");
